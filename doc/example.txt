# 1. mkresp
# 2. mksrc
# 3. mkbg
# 4. simobs
# 5. richlucy_bg


source ~/github/moriiism/mitool/setup/setup_letsnote.sh 

cd /home/morii/work/arb/run

# responseの前提
nskyx=60
nskyy=60
ndetx=60
ndety=60

# 1. make response matrix, normed response matrix, 
#    and efficiency matrix files

respdir=../data/imageall
outdir=resp
outfile_head=arb

~/github/moriiism/srt/mkresp/mkresp \
$respdir \
$outdir \
$outfile_head

--->
  resp/arb_eff.fits        : efficiency file
  resp/arb_resp.fits       : response file
  resp/arb_resp_norm.fits  : normalized response file

# 2. mksrc

mkdir sim
cat << EOF > sim/two.dat
# posx posy flux
15 15 1.0
20 20 1.0
EOF

infile=sim/two.dat
outfile=sim/two.fits

~/github/moriiism/srt/mksrc/mksrc \
$infile \
$outfile

--> 
  sim/two.fits

# 3. mkbg

cat << EOF > sim/bg.dat
# background model name
# background model parameters
quad
30 30 -1  0  0 10000
EOF

model_file=sim/bg.dat
outfile=sim/bg.fits

~/github/moriiism/srt/mkbg/mkbg \
$model_file \
$outfile

--> 
  sim/bg.fits

# 4. simobs

respfile=resp/arb_resp.fits
srcfile=sim/two.fits
nevt_src=10000
bgfile=sim/bg.fits
nevt_bg=3000
rand_seed_det=1
rand_seed_partition=1
nfold=1
npartition=1
outdir=sim
outfile_head=two

~/github/moriiism/srt/simobs/simobs \
$respfile \
$srcfile \
$nevt_src \
$bgfile \
$nevt_bg \
$rand_seed_det \
$rand_seed_partition \
$nfold \
$npartition \
$outdir \
$outfile_head

-->
  sim/two_obs_0001.fits  : observation data in detector coordinate
  sim/two_bg_3000.fits   : background model in detector coordinate


# 5. richlucy_bg
